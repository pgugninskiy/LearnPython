/*
================================================================================
ЗАПРОС: Выгрузка текущего плана производства для внешней системы
================================================================================
Цель: Отправка значения текущего плана (current_plan_value) по цеху за дату.
      Показываются ВСЕ записи за дату, включая те, где план = 0.

Параметры:
  :p_date    — Дата, за которую требуется план (обязательно).
               Пример: 2023-10-27
  :p_orgunit — ID подразделения (опционально).
               Если не передан, выгрузятся все записи для цеха '5360'.

Важно:
  - business_plan_value НЕ используется (ни в SELECT, ни в фильтрах).
  - Нулевые значения current_plan_value попадают в выгрузку.
================================================================================
*/
SELECT 
    zpp.id AS id,                           -- Уникальный ID записи плана
    sp.code AS orgunit_code,                -- Код подразделения (цеха)
    sp.short_name AS orgunit_name,          -- Краткое название подразделения
    zpp.plan_date AS plan_date,             -- Дата планового показателя
    zpp.current_plan_value AS current_plan, -- Текущее значение плана (основное)
    pz.production_order_number AS order_number, -- Номер производственного заказа
    spz.name AS order_status,               -- Статус производственного заказа
    to_char(zpp.plan_date, 'DD.MM.YYYY') AS plan_date_str, -- Дата в формате строки (для отчета)
    CONCAT(sp.short_name, '-', zpp.plan_date) AS unique_value -- Уникальный ключ (Имя-Дата)
FROM
    /* Таблица с плановыми значениями */
    "Значения плановых показателей на дату" AS zpp
    /* Справочник подразделений (обязательная связь) */
    INNER JOIN "Структурное подразделение" AS sp
        ON zpp.orgunit = sp.id
    /* Производственный заказ (может отсутствовать) */
    LEFT JOIN "Производственный заказ" AS pz
        ON zpp.production_order = pz.id
    /* Статус заказа (может отсутствовать) */
    LEFT JOIN "Статус производственного заказа" AS spz
        ON pz.production_order_status = spz.id
WHERE
    -- 1. Исключаем логически удаленные записи (защита от мусора)
    sp.deleted = false 
    AND zpp.deleted = false 
    -- 2. Жесткая привязка к цеху '5360'
    -- Если потребуется другой цех, изменить значение здесь
    AND sp.code = '5360' 
    AND zpp.id =:id_plan_indicators
ORDER BY 
    zpp.plan_date  -- Сортировка по дате
